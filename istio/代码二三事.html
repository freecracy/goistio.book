# 创建多维数组

// 使用独立slice创建动态的多维数组,由外到内
table := make([][]int, 3) // 1. 创建一个外部slice
for v := range table {
	table[v] = make([]int, 4) // 2. 分配每个内部slice
}
fmt.Println(table)
// 使用共享数据slice的slice创建多维数组,由内到外
raw := make([]int, 3) // 创建原始数据容器
for v := range raw {
	raw[v] = v
}
table := make([][]int, 4) //创建外部slice
for v := range table {
	table[v] = raw //重新切片原始数据
}
fmt.Println(table)

# 关闭http响应

// 当你的http响应失败时，resp变量将为nil，而err变量将是non-nil。然而，当你得到一个重定向的错误时，两个变量都将是non-nil。这意味着你最后依然会内存泄露
resp, err := http.Get("https://api.ipify.org?format=json")
if resp != nil {
   defer resp.Body.Close()
}
// resp.Body.Close()的原始实现也会读取并丢弃剩余的响应主体数据。这确保了http的链接在keepalive http连接行为开启的情况下，可以被另一个请求复用。最新的http客户端的行为是不同的。现在读取并丢弃剩余的响应数据是你的职责。如果你不这么做，http的连接可能会关闭，而无法被重用
_, err = io.Copy(ioutil.Discard, resp.Body) // http连接的重用

# 关闭http连接

// 标准http库只在目标HTTP服务器要求关闭时才会关闭网络连接。这意味着你的应用在某些条件下消耗完sockets/file的描述符。
// 你可以通过设置请求变量中的Close域的值为true，来让http库在请求完成时关闭连接。
req.Close = true
// 另一个选项是添加一个Connection的请求头，并设置为close。目标HTTP服务器应该也会响应一个Connection: close的头。当http库看到这个响应头时，它也将会关闭连接。
req.Header.Add("Connection", "close")

# .(type)

type T struct {
}

// .(type) 用来类型分支选择上
func printType(v interface{}) {
	switch v.(type) {
	case int:
		fmt.Println("int")
	case T:
		fmt.Println("T")
	}
}

func main() {
	var v interface{} = 10
	_, ok := v.(int) // 检查v是不是int,ok:[ture|false]
	if ok {
		fmt.Println(v.(int)) //.(SomeType)是把一个接口转换成具体的类型,如果v不是int会panic
	}
}

# recover

// recover只有在defer函数中才有用,在defer函数调用的函数中recover不起作用
// 一般的panic都能被recover捕获,slice越界,nil指针,除0,写关闭的channel
// 一般标准库中通过throw抛出的错误都是无法recover的
// 超过线程数限制的panic时无法捕获的,默认1w个物理线程
func main() {
	f := func() {
		if r := recover(); r != nil {
			println(r)
		}
	}
	defer func() {
		f()
	}()
	panic("ok")
}
// defer 一般用在资源释放,或者捕获panic
// panic 时终止正常的执行流程,执行所有defer,
// recover 在panic时获取控制权,进入正常执行逻辑

# defer

// defer在声明时引用到的变量就已经被实时的编译了
var ErrNotFound error = errors.New("Not found")

func TestDefer1() error {
	var err error
	defer fmt.Printf("err:%v\n", err) // err:<nil>
	err = ErrNotFound
	return err
}

func TestDefer2() error {
	var err error
	defer func() {
		fmt.Printf("err:%v\n", err) // err:Not found
	}()
	err = ErrNotFound
	return err
}
