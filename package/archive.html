const (
	TypeReg  = '0' //普通文件
	TypeRegA = '\x00' // 普通文件,不推荐使用

	// 有header,没有body的文件
	TypeLink    = '1' // 硬连接
	TypeSymlink = '2' // 软连接
	TypeChar    = '3' // 字符设备节点
	TypeBlock   = '4' // 块设备节点
	TypeDir     = '5' // 目录
	TypeFifo    = '6' // fifo节点

	TypeCont = '7' // 保留字段
	TypeXHeader = 'x' // 扩展标题
	TypeXGlobalHeader = // 全局扩展标题

	TypeGNUSparse = 'S' // 稀疏文件
	TypeGNULongName = 'L' //下一个文件名称很长
	TypeGNULongLink = 'K' //将文件符号连接到一个带有长文件名的文件
)

type Header struct {
	Typeflag byte // 标题条目的类型

	Name     string // 文件名称
	Linkname string // 连接的目标名称

	Size  int64  // 以字节为单位的长度
	Mode  int64  // 权限位
	Uid   int    // 所有者用户ID
	Gid   int    // 所有者组ID
	Uname string // 所有者用户名
	Gname string // 所有者组名

	ModTime    time.Time // 权限修改时间
	AccessTime time.Time // 访问时间
	ChangeTime time.Time // 文件修改时间

	Devmajor int64 // 主要块设备数量
	Devminor int64 // 次要块设备数量

	Xattrs map[string]string
	PAXRecords map[string]string

	Format Format
}
